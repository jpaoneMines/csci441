643inline
644bool CSCI441::OpenGLEngine::saveScreenshot(
645 const char* FILENAME
646) noexcept {
647 // Generate a name based on current timestamp if not provided
648 const std::string filename =
649 (
650 FILENAME == nullptr
651 ? "Screenshot_" + std::to_string(time(nullptr)) + ".png"
652 : FILENAME
653 );
654
655 // Get size
656 GLint viewport[4];
657 glGetIntegerv(GL_VIEWPORT, viewport);
658 const GLsizei x = viewport[0], y = viewport[1], width = viewport[2], height = viewport[3];
659
660 // Read pixel data
661 constexpr int CHANNELS = 4; // RGBA
662 const auto bytes = new GLubyte[width*height*CHANNELS];
663 glReadPixels(x, y, width, height, GL_RGBA, GL_UNSIGNED_BYTE, bytes);
664
665 stbi_flip_vertically_on_write(true);
666
667 if( !stbi_write_png(filename.c_str(), width, height, CHANNELS, bytes, width*CHANNELS) ) {
668 fprintf(stderr, "[ERROR]: Could not save screenshot\n");
669 mErrorCode = OPENGL_ENGINE_ERROR_TAKE_SCREENSHOT;
670 } else if(DEBUG) {
671 fprintf(stdout, "[INFO]: Screenshot saved to %s\n", filename.c_str());
672 }
673
674 delete[] bytes;
675
676 return (mErrorCode == OPENGL_ENGINE_ERROR_NO_ERROR);
677}

679inline
680void CSCI441::OpenGLEngine::mSetupGLFW()
681{
682 // set what function to use when registering errors
683 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
684 // all other GLFW calls must be performed after GLFW has been initialized
685 glfwSetErrorCallback(mErrorCallback);
686
687 // initialize GLFW
688 if( !glfwInit() ) {
689 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
690 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
691 } else {
692 if(DEBUG) fprintf( stdout, "[INFO]: GLFW v%d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
693
694 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, mOpenGLMajorVersion ); // request OpenGL vX.
695 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, mOpenGLMinorVersion ); // request OpenGL v .X
696 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
697 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
698 glfwWindowHint(GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
699 glfwWindowHint(GLFW_RESIZABLE, mWindowResizable ); // set if our window should be able to be resized
700
701 // if wanting debug information with Version 4.3 or higher
702 if( DEBUG
703 && (mOpenGLMajorVersion > 4 || (mOpenGLMajorVersion == 4 && mOpenGLMinorVersion >= 3)) ) {
704 glfwWindowHint(GLFW_OPENGL_DEBUG_CONTEXT, true); // request a debug context
705 }
706
707 // create a window for a given size, with a given title
708 mpWindow = glfwCreateWindow(mWindowWidth, mWindowHeight, mWindowTitle, nullptr, nullptr );
709 if( !mpWindow ) { // if the window could not be created, NULL is returned
710 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
711 glfwTerminate();
712 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
713 } else {
714 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
715 glfwMakeContextCurrent(mpWindow); // make the created window the current window
716 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
717 glfwSetInputMode(mpWindow, GLFW_LOCK_KEY_MODS, GLFW_TRUE); // track state of Caps Lock and Num Lock keys
718 glfwSetWindowUserPointer(mpWindow, (void*)this);
719 glfwSetWindowSizeCallback(mpWindow, mWindowResizeCallback);
720 }
721 }

724inline
725void CSCI441::OpenGLEngine::_setupGLFunctions()
726{
727
728#ifdef CSCI441_USE_GLEW
729 glewExperimental = GL_TRUE;
730 const GLenum glewResult = glewInit(); // initialize GLEW
731
732 // check for an error
733 if( glewResult != GLEW_OK ) {
734 fprintf( stderr, "[ERROR]: Error initializing GLEW\n");
735 fprintf( stderr, "[ERROR]: %s\n", reinterpret_cast<const char *>(glewGetErrorString(glewResult)) );
736 mErrorCode = OPENGL_ENGINE_ERROR_GLEW_INIT;
737 } else {
738 if(DEBUG) {
739 fprintf(stdout, "[INFO]: GLEW initialized\n");
740 fprintf(stdout, "[INFO]: Using GLEW %s\n", reinterpret_cast<const char *>(glewGetString(GLEW_VERSION)));
741 }
742 }
743#else
744 int version = gladLoadGL(glfwGetProcAddress);
745 if(version == 0) {
746 fprintf(stderr, "[ERROR]: Failed to initialize GLAD\n" );
747 mErrorCode = OPENGL_ENGINE_ERROR_GLAD_INIT;
748 } else {
749 if(DEBUG) {
750 // Successfully loaded OpenGL
751 fprintf(stdout, "[INFO]: GLAD initialized v%d.%d\n", GLAD_VERSION_MAJOR(version), GLAD_VERSION_MINOR(version));
752 }
753 }
754#endif
755
756 if(mErrorCode == OPENGL_ENGINE_ERROR_NO_ERROR) {
757 glGetIntegerv(GL_MAJOR_VERSION, &mOpenGLMajorVersionCreated);
758 glGetIntegerv(GL_MINOR_VERSION, &mOpenGLMinorVersionCreated);
759 if(DEBUG) {
760 fprintf(stdout, "[INFO]: OpenGL v%d.%d requested\n", mOpenGLMajorVersionRequested, mOpenGLMinorVersionRequested);
761 fprintf(stdout, "[INFO]: OpenGL v%d.%d created\n", mOpenGLMajorVersionCreated, mOpenGLMinorVersionCreated);
762 }
763
764 GLint numExtensions = 0;
765 glGetIntegerv(GL_NUM_EXTENSIONS, &numExtensions);
766 for (int i = 0; i < numExtensions; i++) {
767 _extensions.insert(reinterpret_cast<const char *>(glGetStringi(GL_EXTENSIONS, i)) );
768 }
769 }
