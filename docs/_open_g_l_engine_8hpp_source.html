512inline
513void CSCI441::OpenGLEngine::initialize()
514{
515 if( !_isInitialized ) {
516 if (DEBUG) {
517 fprintf(stdout, "[INFO]: Using CSCI441 Library v%d.%d.%d\n", CSCI441::VERSION_MAJOR, CSCI441::VERSION_MINOR, CSCI441::VERSION_PATCH);
518 }
519
520 mSetupGLFW(); // initialize GLFW and set up a window
521 _setupGLFunctions(); // create OpenGL function pointers
522 mSetupOpenGL(); // create the OpenGL context
523
524 // get OpenGL context information
525 if( DEBUG ) {
526 // if wanting debug information with Version 4.3 or higher
527 if( mOpenGLMajorVersion > 4 || (mOpenGLMajorVersion == 4 && mOpenGLMinorVersion >= 3) ) {
528 // check if debug context was created
529 int flags; glGetIntegerv(GL_CONTEXT_FLAGS, &flags);
530 if (flags & GL_CONTEXT_FLAG_DEBUG_BIT) {
531 // register callback to synchronously print any debug messages without having to call glGetError()
532 glEnable(GL_DEBUG_OUTPUT);
533 glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS);
534 glDebugMessageCallback(mDebugMessageCallback, nullptr);
535 glDebugMessageControl( GL_DONT_CARE, GL_DONT_CARE, GL_DONT_CARE, 0, nullptr, GL_TRUE );
536 }
537 }
538
539 CSCI441::OpenGLUtils::printOpenGLInfo();
540 }
541
542 mSetupShaders(); // transfer, compile, link shaders on GPU
543 mSetupBuffers(); // register Buffers on GPU
544 mSetupTextures(); // register Textures on GPU
545 mSetupScene(); // setup any scene specific information
546
547 _isInitialized = true;
548 _isCleanedUp = false;
549 if (DEBUG) fprintf(stdout, "\n[INFO]: Setup complete\n");
550 }

568inline
569bool CSCI441::OpenGLEngine::saveScreenshot(
570 const char* FILENAME
571) noexcept {
572 // Generate a name based on current timestamp if not provided
573 const std::string filename =
574 (
575 FILENAME == nullptr
576 ? "Screenshot_" + std::to_string(time(nullptr)) + ".png"
577 : FILENAME
578 );
579
580 // Get size
581 GLint viewport[4];
582 glGetIntegerv(GL_VIEWPORT, viewport);
583 const GLsizei x = viewport[0], y = viewport[1], width = viewport[2], height = viewport[3];
584
585 // Read pixel data
586 constexpr int CHANNELS = 4; // RGBA
587 const auto bytes = new GLubyte[width*height*CHANNELS];
588 glReadPixels(x, y, width, height, GL_RGBA, GL_UNSIGNED_BYTE, bytes);
589
590 stbi_flip_vertically_on_write(true);
591
592 if( !stbi_write_png(filename.c_str(), width, height, CHANNELS, bytes, width*CHANNELS) ) {
593 fprintf(stderr, "[ERROR]: Could not save screenshot\n");
594 mErrorCode = OPENGL_ENGINE_ERROR_TAKE_SCREENSHOT;
595 } else if(DEBUG) {
596 fprintf(stdout, "[INFO]: Screenshot saved to %s\n", filename.c_str());
597 }
598
599 delete[] bytes;
600
601 return (mErrorCode == OPENGL_ENGINE_ERROR_NO_ERROR);
602}

604inline
605void CSCI441::OpenGLEngine::mSetupGLFW()
606{
607 // set what function to use when registering errors
608 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
609 // all other GLFW calls must be performed after GLFW has been initialized
610 glfwSetErrorCallback(mErrorCallback);
611
612 // initialize GLFW
613 if( !glfwInit() ) {
614 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
615 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
616 } else {
617 if(DEBUG) fprintf( stdout, "[INFO]: GLFW %d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
618
619 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, mOpenGLMajorVersion ); // request OpenGL vX.
620 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, mOpenGLMinorVersion ); // request OpenGL v .X
621 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
622 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
623 glfwWindowHint(GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
624 glfwWindowHint(GLFW_RESIZABLE, mWindowResizable ); // set if our window should be able to be resized
625
626 // if wanting debug information with Version 4.3 or higher
627 if( DEBUG
628 && (mOpenGLMajorVersion > 4 || (mOpenGLMajorVersion == 4 && mOpenGLMinorVersion >= 3)) ) {
629 glfwWindowHint(GLFW_OPENGL_DEBUG_CONTEXT, true); // request a debug context
630 }
631
632 // create a window for a given size, with a given title
633 mpWindow = glfwCreateWindow(mWindowWidth, mWindowHeight, mWindowTitle, nullptr, nullptr );
634 if( !mpWindow ) { // if the window could not be created, NULL is returned
635 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
636 glfwTerminate();
637 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
638 } else {
639 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
640 glfwMakeContextCurrent(mpWindow); // make the created window the current window
641 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
642 glfwSetInputMode(mpWindow, GLFW_LOCK_KEY_MODS, GLFW_TRUE); // track state of Caps Lock and Num Lock keys
643 glfwSetWindowUserPointer(mpWindow, (void*)this);
644 glfwSetWindowSizeCallback(mpWindow, mWindowResizeCallback);
645 }
646 }

649inline
650void CSCI441::OpenGLEngine::_setupGLFunctions()
651{
652
653#ifdef CSCI441_USE_GLEW
654 glewExperimental = GL_TRUE;
655 const GLenum glewResult = glewInit(); // initialize GLEW
656
657 // check for an error
658 if( glewResult != GLEW_OK ) {
659 fprintf( stderr, "[ERROR]: Error initializing GLEW\n");
660 fprintf( stderr, "[ERROR]: %s\n", reinterpret_cast<const char *>(glewGetErrorString(glewResult)) );
661 mErrorCode = OPENGL_ENGINE_ERROR_GLEW_INIT;
662 } else {
663 if(DEBUG) {
664 fprintf(stdout, "\n[INFO]: GLEW initialized\n");
665 fprintf(stdout, "[INFO]: Using GLEW %s\n", reinterpret_cast<const char *>(glewGetString(GLEW_VERSION)));
666 }
667 }
668#else
669 int version = gladLoadGL(glfwGetProcAddress);
670 if(version == 0) {
671 fprintf(stderr, "Failed to initialize GLAD\n" );
672 mErrorCode = OPENGL_ENGINE_ERROR_GLAD_INIT;
673 } else {
674 if(DEBUG) {
675 // Successfully loaded OpenGL
676 fprintf(stdout, "\n[INFO]: GLAD initialized\n");
677 fprintf(stdout, "[INFO]: Loaded OpenGL %d.%d\n", GLAD_VERSION_MAJOR(version), GLAD_VERSION_MINOR(version));
678 }
679 }
680#endif
681
682 if(mErrorCode == OPENGL_ENGINE_ERROR_NO_ERROR) {
683 GLint numExtensions = 0;
684 glGetIntegerv(GL_NUM_EXTENSIONS, &numExtensions);
685 for (int i = 0; i < numExtensions; i++) {
686 _extensions.insert(reinterpret_cast<const char *>(glGetStringi(GL_EXTENSIONS, i)) );
687 }
688 }
689}
