282 GLint numExtensions = 0;

284

299inline const char* CSCI441::OpenGLUtils::openGLErrorMessage(GLenum err) {

302 case GL_INVALID_ENUM: return "Invalid enum";

304 case GL_INVALID_OPERATION: return "Invalid operation";
