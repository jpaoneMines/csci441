128inline bool CSCI441::FontUtils::loadFont(const char * const filename) {
129 if (CSCI441_INTERNAL::FontUtils::currentFont != nullptr) {
130 delete CSCI441_INTERNAL::FontUtils::currentFont;
131 CSCI441_INTERNAL::FontUtils::currentFont = nullptr;
132 }
133
134 bool success = true;
135
136 if (CSCI441_INTERNAL::FontUtils::fontShaderHandle == 0) {
137 success &= CSCI441_INTERNAL::FontUtils::loadFontShader();
138 }
139
140 CSCI441_INTERNAL::FontUtils::currentFont = new CSCI441::Font(filename);
141 CSCI441_INTERNAL::FontUtils::currentFont->setScale(CSCI441_INTERNAL::FontUtils::fontSize.x, CSCI441_INTERNAL::FontUtils::fontSize.y);
142
143 return success && CSCI441_INTERNAL::FontUtils::currentFont->isLoaded();;
144}

204inline bool CSCI441_INTERNAL::FontUtils::loadFontShader() {
205 const std::string vertex_shader_src =
206R"_(#version 410 core
207
208layout(location = 0) in vec4 coord;
209
210out vec2 vTexCoord;
211
212void main() {
213 gl_Position = vec4( coord.xy, 0.0f, 1.0f );
214 vTexCoord = coord.zw;
215})_";
216
217 const std::string fragment_shader_src =
218R"_(#version 410 core
219
220uniform sampler2D tex;
221uniform vec4 color;
222
223in vec2 vTexCoord;
224
225out vec4 fragColorOut;
226
227void main() {
228 fragColorOut = vec4(1.0, 1.0, 1.0, texture(tex, vTexCoord).r) * color;
229}
230)_";
231
232 const char* vertexShaders[1] = { vertex_shader_src.c_str() };
233 const char* fragmentShaders[1] = { fragment_shader_src.c_str() };
234
235 const GLuint vertexShaderHandle = glCreateShader( GL_VERTEX_SHADER );
236 glShaderSource(vertexShaderHandle, 1, vertexShaders, nullptr);
237 glCompileShader(vertexShaderHandle);
238
239 const GLuint fragmentShaderHandle = glCreateShader( GL_FRAGMENT_SHADER );
240 glShaderSource(fragmentShaderHandle, 1, fragmentShaders, nullptr);
241 glCompileShader(fragmentShaderHandle);
242
243 fontShaderHandle = glCreateProgram();
244 glAttachShader(fontShaderHandle, vertexShaderHandle);
245 glAttachShader(fontShaderHandle, fragmentShaderHandle);
246 glLinkProgram(fontShaderHandle);
247
248 glDetachShader(fontShaderHandle, vertexShaderHandle);
249 glDeleteShader(vertexShaderHandle);
250
251 glDetachShader(fontShaderHandle, fragmentShaderHandle);
252 glDeleteShader(fragmentShaderHandle);
253
254 const GLint texUniformLocation = glGetUniformLocation(fontShaderHandle, "tex");
255 fontColorUniformLocation = glGetUniformLocation(fontShaderHandle, "color");
256
257 glProgramUniform1i(fontShaderHandle, texUniformLocation, 0);
258 constexpr glm::vec4 white(1.0f, 1.0f, 1.0f, 1.0f);
259 glProgramUniform4fv(fontShaderHandle, fontColorUniformLocation, 1, glm::value_ptr(white));
260
261 return fontShaderHandle != 0;
262
263}
