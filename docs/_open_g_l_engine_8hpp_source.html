472inline void CSCI441::OpenGLEngine::initialize() {
473 if( !_isInitialized ) {
474 mSetupGLFW(); // initialize GLFW and set up a window
475 _setupGLFunctions(); // create OpenGL function pointers
476 mSetupOpenGL(); // create the OpenGL context
477
478 // get OpenGL context information
479 if( DEBUG ) {
480 // if wanting debug information with Version 4.3 or higher
481 if( mOpenGLMajorVersion > 4 || (mOpenGLMajorVersion == 4 && mOpenGLMinorVersion >= 3) ) {
482 // check if debug context was created
483 int flags; glGetIntegerv(GL_CONTEXT_FLAGS, &flags);
484 if (flags & GL_CONTEXT_FLAG_DEBUG_BIT) {
485 // register callback to synchronously print any debug messages without having to call glGetError()
486 glEnable(GL_DEBUG_OUTPUT);
487 glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS);
488 glDebugMessageCallback(mDebugMessageCallback, nullptr);
489 glDebugMessageControl( GL_DONT_CARE, GL_DONT_CARE, GL_DONT_CARE, 0, nullptr, GL_TRUE );
490 }
491 }
492
493 CSCI441::OpenGLUtils::printOpenGLInfo();
494 }
495
496 mSetupShaders(); // transfer, compile, link shaders on GPU
497 mSetupBuffers(); // register Buffers on GPU
498 mSetupTextures(); // register Textures on GPU
499 mSetupScene(); // setup any scene specific information
500
501 _isInitialized = true;
502 _isCleanedUp = false;
503 if (DEBUG) fprintf(stdout, "\n[INFO]: Setup complete\n");
504 }
505}

507inline void CSCI441::OpenGLEngine::mSetupGLFW() {
508 // set what function to use when registering errors
509 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
510 // all other GLFW calls must be performed after GLFW has been initialized
511 glfwSetErrorCallback(mErrorCallback);
512
513 // initialize GLFW
514 if( !glfwInit() ) {
515 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
516 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
517 } else {
518 if(DEBUG) fprintf( stdout, "[INFO]: GLFW %d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
519
520 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, mOpenGLMajorVersion ); // request OpenGL vX.
521 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, mOpenGLMinorVersion ); // request OpenGL v .X
522 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
523 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
524 glfwWindowHint(GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
525 glfwWindowHint(GLFW_RESIZABLE, mWindowResizable ); // set if our window should be able to be resized
526
527 // if wanting debug information with Version 4.3 or higher
528 if( DEBUG
529 && (mOpenGLMajorVersion > 4 || (mOpenGLMajorVersion == 4 && mOpenGLMinorVersion >= 3)) ) {
530 glfwWindowHint(GLFW_OPENGL_DEBUG_CONTEXT, true); // request a debug context
531 }
532
533 // create a window for a given size, with a given title
534 mpWindow = glfwCreateWindow(mWindowWidth, mWindowHeight, mWindowTitle, nullptr, nullptr );
535 if( !mpWindow ) { // if the window could not be created, NULL is returned
536 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
537 glfwTerminate();
538 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
539 } else {
540 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
541 glfwMakeContextCurrent(mpWindow); // make the created window the current window
542 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
543 glfwSetInputMode(mpWindow, GLFW_LOCK_KEY_MODS, GLFW_TRUE); // track state of Caps Lock and Num Lock keys
544 glfwSetWindowUserPointer(mpWindow, (void*)this);
545 glfwSetWindowSizeCallback(mpWindow, mWindowResizeCallback);
546 }
547 }

550inline void CSCI441::OpenGLEngine::_setupGLFunctions() {
551
552#ifdef CSCI441_USE_GLEW
553 glewExperimental = GL_TRUE;
554 const GLenum glewResult = glewInit(); // initialize GLEW
555
556 // check for an error
557 if( glewResult != GLEW_OK ) {
558 fprintf( stderr, "[ERROR]: Error initializing GLEW\n");
559 fprintf( stderr, "[ERROR]: %s\n", reinterpret_cast<const char *>(glewGetErrorString(glewResult)) );
560 mErrorCode = OPENGL_ENGINE_ERROR_GLEW_INIT;
561 } else {
562 if(DEBUG) {
563 fprintf(stdout, "\n[INFO]: GLEW initialized\n");
564 fprintf(stdout, "[INFO]: Using GLEW %s\n", reinterpret_cast<const char *>(glewGetString(GLEW_VERSION)));
565 }
566 }
567#else
568 int version = gladLoadGL(glfwGetProcAddress);
569 if(version == 0) {
570 fprintf(stderr, "Failed to initialize GLAD\n" );
571 mErrorCode = OPENGL_ENGINE_ERROR_GLAD_INIT;
572 } else {
573 if(DEBUG) {
574 // Successfully loaded OpenGL
575 fprintf(stdout, "\n[INFO]: GLAD initialized\n");
576 fprintf(stdout, "[INFO]: Loaded OpenGL %d.%d\n", GLAD_VERSION_MAJOR(version), GLAD_VERSION_MINOR(version));
577 }
578 }
579#endif
580
581 if(mErrorCode == OPENGL_ENGINE_ERROR_NO_ERROR) {
582 GLint numExtensions = 0;
583 glGetIntegerv(GL_NUM_EXTENSIONS, &numExtensions);
584 for (int i = 0; i < numExtensions; i++) {
585 _extensions.insert(reinterpret_cast<const char *>(glGetStringi(GL_EXTENSIONS, i)) );
586 }
587 }
588}

598inline void CSCI441::OpenGLEngine::shutdown() {
599 if( !_isCleanedUp ) {
600 if (DEBUG) fprintf(stdout, "\n[INFO]: Shutting down.......\n");
601 mCleanupShaders(); // delete shaders from GPU
602 mCleanupBuffers(); // delete VAOs/VBOs from GPU
603 mCleanupTextures(); // delete textures from GPU
604 mCleanupScene(); // delete scene info from CPU
605 mCleanupOpenGL(); // cleanup anything OpenGL related
606 _cleanupGLFunctions(); // cleanup anything function pointer related
607 mCleanupGLFW(); // shut down GLFW to clean up our context
608 if (DEBUG) fprintf(stdout, "[INFO]: ..shut down complete!\n");
609 _isCleanedUp = true;
610 _isInitialized = false;
611 }
