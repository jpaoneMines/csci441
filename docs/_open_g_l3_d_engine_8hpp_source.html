156 [[maybe_unused]] virtual void handleCameraCursorPosEvent(const double x, const double y) final {
157 const glm::vec2 currMousePos(x, getWindowHeight() - y);
158
159 if( mIsLeftMouseDown ) {
160 if( !mIsShiftDown ) {
161 // rotate the camera by the distance the mouse moved
162 mpArcballCamera->rotate((currMousePos.x - mMousePosition.x) * 0.005f,
163 (mMousePosition.y - currMousePos.y) * 0.005f);
164 } else {
165 // otherwise, update our camera angles theta & phi
166 const GLfloat totChgSq = (currMousePos.x - mMousePosition.x) + (currMousePos.y - mMousePosition.y);
167 mpArcballCamera->moveForward( totChgSq * 0.05f );
168 }
169 }
170
171 setMousePosition(currMousePos);
172 }
