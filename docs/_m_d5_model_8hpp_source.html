1707inline void
1708CSCI441::MD5Model::_prepareMesh(
1709 const MD5Mesh *pMESH
1710) const {
1711 GLint i, j, k;
1712
1713 // Setup vertex indices
1714 for(k = 0, i = 0; i < pMESH->numTriangles; ++i) {
1715 for(j = 0; j < 3; ++j, ++k)
1716 _vertexIndicesArray[k] = pMESH->triangles[i].index[j];
1717 }
1718
1719 // Setup vertices
1720 for(i = 0; i < pMESH->numVertices; ++i) {
1721 glm::vec3 finalVertex = {0.0f, 0.0f, 0.0f };
1722
1723 // Calculate final vertex to draw with weights
1724 for(j = 0; j < pMESH->vertices[i].count; ++j) {
1725 const MD5Weight *weight = &pMESH->weights[pMESH->vertices[i].start + j];
1726 const MD5Joint *joint = &_skeleton[weight->joint];
1727
1728 // Calculate transformed vertex for this weight
1729 const glm::vec3 weightedVertex = glm::rotate(joint->orientation, glm::vec4(weight->position, 0.0f));
1730
1731 // The sum of all weight->bias should be 1.0
1732 finalVertex.x += (joint->position.x + weightedVertex.x) * weight->bias;
1733 finalVertex.y += (joint->position.y + weightedVertex.y) * weight->bias;
1734 finalVertex.z += (joint->position.z + weightedVertex.z) * weight->bias;
1735 }
1736
1737 _vertexArray[i].x = finalVertex.x;
1738 _vertexArray[i].y = finalVertex.y;
1739 _vertexArray[i].z = finalVertex.z;
1740
1741 _texelArray[i].s = pMESH->vertices[i].texCoord.s;
1742 _texelArray[i].t = pMESH->vertices[i].texCoord.t;
1743 }
1744
1745 glBindVertexArray(_vao );
1746
1747 glBindBuffer(GL_ARRAY_BUFFER, _vbo[0] );
1748 glBufferSubData(GL_ARRAY_BUFFER, 0, static_cast<GLsizeiptr>(sizeof(glm::vec3)) * pMESH->numVertices, &_vertexArray[0] );
1749 glBufferSubData(GL_ARRAY_BUFFER, static_cast<GLsizeiptr>(sizeof(glm::vec3)) * _maxVertices, static_cast<GLsizeiptr>(sizeof(glm::vec2)) * pMESH->numVertices, &_texelArray[0] );
1750
1751 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, _vbo[1] );
1752 glBufferSubData(GL_ELEMENT_ARRAY_BUFFER, 0, static_cast<GLsizeiptr>(sizeof(GLuint)) * pMESH->numTriangles * 3, _vertexIndicesArray );
1753}
