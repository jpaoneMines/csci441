58inline void CSCI441::FramebufferUtils::printFramebufferInfo( const GLenum target, const GLuint fbo ) {
59 int res, i = 0;
60 GLint buffer;
61
62 if(glIsFramebuffer(fbo)) {
63 glBindFramebuffer( target, fbo );
64
65 do {
66 glGetIntegerv( GL_DRAW_BUFFER0 + i, &buffer );
67 if( buffer != GL_NONE ) {
68 printf( "[FBO]: Shader Output Location %d -> color attachment %d\n", i, buffer - GL_COLOR_ATTACHMENT0 );
69 glGetFramebufferAttachmentParameteriv( target, buffer, GL_FRAMEBUFFER_ATTACHMENT_OBJECT_TYPE, &res );
70 printf( "[FBO]: \tAttachment Type: %s\n", res == GL_TEXTURE ? "Texture" : "Render Buffer" );
71 glGetFramebufferAttachmentParameteriv( target, buffer, GL_FRAMEBUFFER_ATTACHMENT_OBJECT_NAME, &res );
72 printf( "[FBO]: \tAttachment object name: %d\n", res );
73 }
74 ++i;
75 } while( buffer != GL_NONE );
76 } else {
77 fprintf(stderr, "[FBO]: Error: %d is not a framebuffer\n", fbo);
78 }
79}

91inline void CSCI441::FramebufferUtils::printFramebufferStatusMessage( const GLenum target ) {
92 GLenum status = glCheckFramebufferStatus( target );
93 if( status == GL_FRAMEBUFFER_COMPLETE )
94 printf( "[FBO]: Framebuffer initialized completely!\n" );
95 else {
96 fprintf( stderr, "[FBO]: Framebuffer failed to initialize completely 0x%x.\n", status );
97
98 switch(status) {
99 case GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT:
100 fprintf(stderr, "[FBO]: An attachment could not be bound to framebuffer object!\n");
101 break;
102
103 case GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT:
104 fprintf(stderr, "[FBO]: Attachments are missing! At least one image (texture) must be bound to the framebuffer object!\n");
105 break;
106
107 case GL_FRAMEBUFFER_INCOMPLETE_DIMENSIONS_EXT:
108 fprintf(stderr, "[FBO]: The dimensions of the buffers attached to the currently used framebuffer object do not match!\n");
109 break;
110
111 case GL_FRAMEBUFFER_INCOMPLETE_FORMATS_EXT:
112 fprintf(stderr, "[FBO]: The formats of the currently used framebuffer object are not supported or do not fit together!\n");
113 break;
114
115 case GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER:
116 fprintf(stderr, "[FBO]: A Draw buffer is incomplete or undefined. All draw buffers must specify attachment points that have images attached.\n");
117 break;
118
119 case GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER:
120 fprintf(stderr, "[FBO]: A Read buffer is incomplete or undefined. All read buffers must specify attachment points that have images attached.\n");
121 break;
122
123 case GL_FRAMEBUFFER_INCOMPLETE_MULTISAMPLE:
124 fprintf(stderr, "[FBO]: All images must have the same number of multisample samples.\n");
125 break;
126
127 case GL_FRAMEBUFFER_INCOMPLETE_LAYER_TARGETS :
128 fprintf(stderr, "[FBO]: If a layered image is attached to one attachment, then all attachments must be layered attachments. The attached layers do not have to have the same number of layers, nor do the layers have to come from the same kind of texture.\n");
129 break;
130
131 case GL_FRAMEBUFFER_UNSUPPORTED:
132 fprintf(stderr, "[FBO]: Attempt to use an unsupported format combination!\n");
133 break;
134
135 default:
136 fprintf(stderr, "[FBO]: Unknown error while attempting to create framebuffer object!\n");
137 break;
138 }
139 }
